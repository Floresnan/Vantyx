        .all()

            "etapa": l.etapa,
            "score": 0,
            "interesse": l.interesse,

        "date": "today",

            "leads_hot": 0,
        },
        "top_leads": top_leads,
        "notes": "Modo seguro estabilizado",

@app.post("/leads/upsert", response_model=LeadOut)
def upsert_lead(payload: LeadUpsertIn, db: Session = Depends(get_db)):
    def to_int(v):
        if v is None:
            return None
        if isinstance(v, int):
            return v
        s = str(v).strip()
        if s == "":
            return None
        try:
            return int(float(s))
        except:
            return None

    def to_decimal(v):
        if v is None:
            return None
        if isinstance(v, (int, float, Decimal)):
            try:
                return Decimal(str(v))
            except:
                return None
        s = str(v).strip()
        if s == "":
            return None
        # aceita "R$ 850.000,00" e "850000" e "850.000,00"
        s = (
            s.replace("R$", "")
             .replace("r$", "")
             .replace(" ", "")
             .replace(".", "")
             .replace(",", ".")
        )
        try:
            return Decimal(s)
        except (InvalidOperation, ValueError):
            return None

    def to_date(v):
        if v is None:
            return None
        if isinstance(v, date):
            return v
        s = str(v).strip()
        if s == "":
            return None
        for fmt in ("%Y-%m-%d", "%d/%m/%Y", "%d-%m-%Y"):
            try:
                return datetime.strptime(s, fmt).date()
            except:
                pass
        return None

    lead: Lead | None = None

    # Normaliza tipos (CSV-friendly)
    interesse = to_int(getattr(payload, "interesse", None))
    urgencia = to_int(getattr(payload, "urgencia", None))
    budget = to_decimal(getattr(payload, "budget", None))
    ultimo_contato = to_date(getattr(payload, "ultimo_contato", None))

    meta_leadgen_id = getattr(payload, "meta_leadgen_id", None)
    telefone = getattr(payload, "telefone", None)

    # 1) chave forte
    if meta_leadgen_id:
        lead = db.query(Lead).filter(Lead.meta_leadgen_id == meta_leadgen_id).first()

    # 2) fallback
    if not lead and telefone:
        lead = db.query(Lead).filter(Lead.telefone == telefone).first()

    # cria
    if not lead:
        lead = Lead(
            nome=(payload.nome or "CSV_LEAD"),
            telefone=telefone,
            email=getattr(payload, "email", None),
            origem=(payload.origem or "a6"),
            perfil=getattr(payload, "perfil", None),
            budget=budget,
            etapa=(payload.etapa or "Novo Lead"),
            interesse=interesse,
            urgencia=urgencia,
            ultimo_contato=ultimo_contato,
            meta_form_id=getattr(payload, "meta_form_id", None),
            meta_leadgen_id=meta_leadgen_id,
            campaign_id=getattr(payload, "campaign_id", None),
            campaign_name=getattr(payload, "campaign_name", None),
            adset_id=getattr(payload, "adset_id", None),
            adset_name=getattr(payload, "adset_name", None),
            ad_id=getattr(payload, "ad_id", None),
            ad_name=getattr(payload, "ad_name", None),
            raw=getattr(payload, "raw", None),
        )
        try:
            db.add(lead)
            db.commit()
            db.refresh(lead)
            return lead
        except Exception as e:
            db.rollback()
            raise HTTPException(status_code=422, detail=f"DB error on insert: {e}")

    # atualiza (parcial)
    data = payload.model_dump(exclude_unset=True) if hasattr(payload, "model_dump") else payload.dict(exclude_unset=True)

    # sobrescreve com versões normalizadas
    data["interesse"] = interesse
    data["urgencia"] = urgencia
    data["budget"] = budget
    data["ultimo_contato"] = ultimo_contato

    for k, v in data.items():
        if k == "id":
            continue
        setattr(lead, k, v)

    try:
        db.add(lead)
        db.commit()
        db.refresh(lead)
        return lead
    except Exception as e:
        db.rollback()
        raise HTTPException(status_code=422, detail=f"DB error on update: {e}")

import os
import typer
import httpx
from rich import print
from rich.pretty import Pretty

app = typer.Typer(help="Vantyx CLI (vx)")


def api_base() -> str:
    return os.getenv("VANTYX_API", "http://127.0.0.1:8000")


@app.command()

    """Testa comunicação com a API."""
    base = api_base()
    r = httpx.get(f"{base}/health", timeout=5)
    print(f"[green]OK[/green] {base}/health -> {r.status_code}")
    print(Pretty(r.json()))

@app.command("top")
def top(
    min_score: int = typer.Option(10, "--min-score", help="Score mínimo (ignorando se --no-score-filter)"),
    limit: int = typer.Option(10, "--limit", help="Quantidade máxima"),
    phone_only: bool = typer.Option(False, "--phone-only", help="Mostra apenas leads com telefone"),
    no_score_filter: bool = typer.Option(False, "--no-score-filter", help="Não filtra por score"),
):
    """Mostra os melhores leads (com filtros separados)."""
    base = api_base()
    url = f"{base}/ops/daily"
    try:
        r = httpx.get(url, timeout=10)
        print(f"[cyan]{url}[/cyan] -> {r.status_code}")
        data = r.json()
    except Exception as e:
        print(f"[red]Falha[/red] ao chamar {url}")
        print(str(e))
        raise typer.Exit(code=1)

    leads = data.get("top_leads", []) or []

    selected = []
    for l in leads:
        if phone_only and not l.get("telefone"):
            continue
        if not no_score_filter and (l.get("score") or 0) < min_score:
            continue
        selected.append(l)

    selected = selected[:limit]

    print(f"[bold]TOP[/bold] (min_score={min_score}, limit={limit}, phone_only={phone_only}, no_score_filter={no_score_filter})")
    print(Pretty(selected))

@app.command("import-a6")
def import_a6(
    path: str = typer.Argument(...),
    limit: int = typer.Option(0, "--limit"),
):
    """Importa leads_a6.csv sem duplicar."""
    import csv
    import hashlib

    base = api_base()
    url = f"{base}/leads/upsert"
    p = os.path.expanduser(path)

    ok = 0
    fail = 0

    def to_int(v):
        if not v:
            return None
        try:
            return int(float(str(v).strip()))
        except:
            return None

    def to_float(v):
        if not v:
            return None
        try:
            s = str(v).strip().replace(".", "").replace(",", ".")
            return float(s)
        except:
            return None

    with open(p, newline="", encoding="utf-8") as f:
        reader = csv.DictReader(f)

        for i, row in enumerate(reader, start=1):
            if limit and i > limit:
                break

            nome = (row.get("nome") or "").strip() or None
            origem = (row.get("origem") or "").strip() or "a6"
            perfil = (row.get("perfil") or "").strip() or None
            etapa = (row.get("etapa") or "").strip() or None

            interesse = to_int(row.get("interesse"))
            urgencia = to_int(row.get("urgencia"))
            budget = to_float(row.get("budget"))

            ultimo_contato = (row.get("ultimo_contato") or "").strip() or None

            payload = {
                "nome": nome,
                "origem": origem,
                "perfil": perfil,
                "budget": budget,
                "etapa": etapa,
                "interesse": interesse,
                "urgencia": urgencia,
                "ultimo_contato": ultimo_contato,
                "raw": {"a6_csv": row},
            }

            key_src = "|".join([
                str(nome or ""),
                str(origem or ""),
                str(perfil or ""),
                str(budget or ""),
                str(etapa or ""),
                str(interesse or ""),
                str(urgencia or ""),
                str(ultimo_contato or ""),
            ])

            payload["meta_leadgen_id"] = "a6:" + hashlib.sha1(
                key_src.encode("utf-8")
            ).hexdigest()

            try:
                r = httpx.post(url, json=payload, timeout=15)
                if r.status_code >= 300:
                    fail += 1
                else:
                    ok += 1
            except Exception:
                fail += 1
